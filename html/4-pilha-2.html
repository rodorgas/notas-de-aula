Problema: Dada uma expressão aritmética com +, -, *, ( e ), imprima a expressão correspondente na notação posfixa.

int operador (char c) {
    if (c == '+' || c == '-' || c == '*' || c == '/')
        return 1;
    return 0;
}

int prec(char c) {
    switch(c) {
        case '(': return 0;
        case '+':
        case '-': return 1;
        case '*':
        case '/': return 2;
        default: return -1;
    }
}
